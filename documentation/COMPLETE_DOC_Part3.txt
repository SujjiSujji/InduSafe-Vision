```python
def _detection_loop(self):
    """Continuous frame processing at 30 FPS"""
    frame_count = 0
    
    while self.is_running:
        try:
            if not self.camera_enabled:
                # Display paused message
                frame = np.zeros((480, 640, 3), dtype=np.uint8)
                cv2.putText(frame, "Camera Paused", (200, 240),
                           cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 255, 255), 2)
                self.current_display_frame = frame
                time.sleep(0.1)
                continue
            
            # Capture frame
            frame = self.camera.get_frame()
            
            if frame is None:
                time.sleep(0.01)
                continue
            
            # Update evidence buffer
            self.evidence.update_buffer(frame)
            
            # Detect persons and violations
            detections = self.detector.detect(frame)
            
            # Scan QR codes
            qr_detections = self.qr_scanner.scan_frame(frame)
            
            # Identify workers
            identified_workers = []
            for qr in qr_detections:
                worker_id = self.qr_scanner.extract_worker_id(qr['data'])
                worker = self.worker_model.get_by_qr(worker_id)
                if worker:
                    identified_workers.append({
                        'worker': worker,
                        'qr_bbox': qr['rect']
                    })
            
            # Handle violations
            if detections['violations']:
                self._handle_violations(
                    frame, detections, identified_workers, frame_count
                )
            
            # Create annotated display
            display_frame = self._create_display_frame(
                frame, detections, qr_detections, identified_workers
            )
            
            self.current_display_frame = display_frame
            frame_count += 1
            time.sleep(0.01)  # ~30 FPS
            
        except Exception as e:
            print(f"Error: {e}")
            time.sleep(0.1)
```

Sample 2: Violation Handling with Threshold Logic
```python
def _handle_violations(self, frame, detections, workers, frame_count):
    """Threshold-based false positive reduction"""
    threshold = 3  # VIOLATION_THRESHOLD
    max_gap = 10  # Max frames between detections
    
    for violation in detections['violations']:
        # Match to nearest worker
        matched_worker = self._match_worker(violation, workers)
        worker_key = matched_worker['employee_id'] if matched_worker else 'unknown'
        
        # Check continuity
        last_frame = self.last_frame_with_violation.get(worker_key, 0)
        if frame_count - last_frame > max_gap:
            self.violation_counters[worker_key] = 0
        
        # Increment counter
        self.last_frame_with_violation[worker_key] = frame_count
        self.violation_counters[worker_key] += 1
        count = self.violation_counters[worker_key]
        
        # Wait for threshold
        if count < threshold:
            print(f"Counting: {count}/{threshold}")
            continue
        
        # Threshold reached!
        self.violation_counters[worker_key] = 0
        
        # Cooldown check
        cooldown = (time.time() - self.recent_violations.get(worker_key, 0)) < 2
        
        # ALWAYS trigger voice alert
        self.voice_alert.alert(
            worker_name=matched_worker['name'] if matched_worker else 'Unknown',
            violation_type=violation['type']
        )
        
        # Save to DB only if cooldown expired
        if not cooldown:
            # Capture evidence
            evidence = self.evidence.capture_violation(
                frame, matched_worker['name'], violation['type']
            )
            
            # Save violation
            violation_id = self.violation_model.create(
                worker_id=matched_worker['id'],
                violation_type=violation['type'],
                image_path=evidence['image_filename'],
                confidence=violation['confidence']
            )
            
            # Create alert record
            self.alert_model.create(
                worker_id=matched_worker['id'],
                violation_id=violation_id,
                alert_message=f"{violation['type']} detected"
            )
            
            self.recent_violations[worker_key] = time.time()
```

Sample 3: QR Code Scanner (qr_scanner.py)
```python
class QRScanner:
    def __init__(self):
        self.detector = cv2.QRCodeDetector()
    
    def scan_frame(self, frame):
        """Detect and decode multiple QR codes"""
        detections = []
        
        try:
            ret, decoded_info, points, _ = \
                self.detector.detectAndDecodeMulti(frame)
            
            if ret and decoded_info:
                for i, data in enumerate(decoded_info):
                    if data:
                        bbox = points[i]
                        x = min([int(p[0]) for p in bbox])
                        y = min([int(p[1]) for p in bbox])
                        w = max([int(p[0]) for p in bbox]) - x
                        h = max([int(p[1]) for p in bbox]) - y
                        
                        detections.append({
                            'data': data,
                            'rect': {'x': x, 'y': y, 'width': w, 'height': h},
                            'polygon': [(int(p[0]), int(p[1])) for p in bbox]
                        })
        except Exception as e:
            pass  # Fallback or silent failure
        
        return detections
    
    def extract_worker_id(self, qr_data):
        """Parse worker ID from QR format"""
        if not qr_data:
            return None
        
        qr_data = qr_data.strip()
        
        if qr_data.upper().startswith('WORKER:'):
            return qr_data.split(':', 1)[1].strip()
        
        return qr_data
```

class EmailReporter:
    def __init__(self, config):
        self.smtp_server = config.SMTP_SERVER
        self.smtp_port = config.SMTP_PORT
        self.username = config.SMTP_USERNAME
        self.password = config.SMTP_PASSWORD
        self.supervisor_email = config.SUPERVISOR_EMAIL
        self.enabled = bool(self.username and self.password)
    
    def send_session_report(self, session_start, violations, workers):
        """Generate and send supervisor report"""
        session_end = datetime.now()
        duration = session_end - session_start
        
        html = f"""
        <html><body>
        <h1>Session Report</h1>
        <p>Duration: {duration}</p>
        <p>Total Violations: {len(violations)}</p>
        <table>
            <tr><th>Time</th><th>Worker</th><th>Type</th></tr>
        """
        
        for v in violations[:50]:
            html += f"""
            <tr>
                <td>{v['timestamp']}</td>
                <td>{v.get('worker_name', 'Unknown')}</td>
                <td>{v['violation_type'].replace('_', ' ').title()}</td>
            </tr>
            """
        
        html += "</table></body></html>"
        
        if self.enabled:
            self.send_email(
                self.supervisor_email,
                f"Session Report {datetime.now().date()}",
                html
            )
    
    def send_worker_reports(self, workers, violations):
        """Send individual reports to workers"""
        results = {'sent': 0, 'failed': 0, 'skipped': 0}
        
        for worker in workers:
            worker_violations = [v for v in violations 
                               if v.get('worker_id') == worker['id']]
            
            if not worker_violations or not worker.get('email'):
                results['skipped'] += 1
                continue
            
            html = self.generate_worker_html(worker, worker_violations)
            subject = f"Safety Report - {worker['name']}"
            
            if self.send_email(worker['email'], subject, html):
                results['sent'] += 1
            else:
                results['failed'] += 1
        
        return results
```

@app.route('/api/live-feed')
def live_feed():
    """MJPEG stream for real-time monitoring"""
    def generate():
        while True:
            if detection_system.current_display_frame is not None:
                frame = detection_system.current_display_frame
                
                ret, buffer = cv2.imencode('.jpg', frame)
                if ret:
                    yield (b'--frame\r\n'
                           b'Content-Type: image/jpeg\r\n\r\n' + 
                           buffer.tobytes() + b'\r\n')
            
            time.sleep(0.033)  # 30 FPS
    
    return Response(generate(),
                   mimetype='multipart/x-mixed-replace; boundary=frame')

class Violation:
    def __init__(self, db):
        self.db = db
    
    def create(self, worker_id, violation_type, image_path=None, 
               zone='Main Floor', confidence=0.0):
        """Save violation to database"""
        conn = self.db.get_connection()
        cursor = conn.cursor()
        
        cursor.execute('''
            INSERT INTO violations 
            (worker_id, violation_type, image_path, zone, confidence)
            VALUES (?, ?, ?, ?, ?)
        ''', (worker_id, violation_type, image_path, zone, confidence))
        
        violation_id = cursor.lastrowid
        conn.commit()
        conn.close()
        return violation_id
    
    def get_stats(self):
        """Get violation statistics"""
        conn = self.db.get_connection()
        cursor = conn.cursor()
        
        cursor.execute('SELECT COUNT(*) FROM violations')
        total = cursor.fetchone()[0]
        
        cursor.execute("""
            SELECT COUNT(*) FROM violations 
            WHERE date(timestamp) = date('now')
        """)
        today = cursor.fetchone()[0]
        
        cursor.execute("""
            SELECT zone, COUNT(*) as count 
            FROM violations GROUP BY zone
        """)
        zones = [dict(row) for row in cursor.fetchall()]
        
        conn.close()
        
        return {
            'total': total,
            'today': today,
            'zones': zones
        }
```
